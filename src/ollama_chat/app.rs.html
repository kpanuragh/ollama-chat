77    pub fn get_config_path(&mut self) -> PathBuf {
78        let mut path = config_dir().expect("Could not find config directory");
79        path.push("ollama-chat"); // App-specific directory
80        path.push("config.json"); // Config file name
81        path
82    }
83
84    /// Asynchronously reads the configuration from the config file.
85    pub async fn read_config(&mut self) -> Result<bool, Box<dyn std::error::Error>> {
86        let path = self.get_config_path();
87
88        // Check if the file exists asynchronously
89        if fs::metadata(&path).await.is_err() {
90            return Err("Config file does not exist".into());
91        }
92
93        let content = fs::read_to_string(&path).await?;
94        self.config = serde_json::from_str(&content)?;
95        Ok(true)
96    }
97
98    /// Asynchronously writes the config file to the default system path.
99    pub async fn write_config(&mut self) -> std::io::Result<()> {
100        let path = self.get_config_path();
101
102        // Ensure the directory exists asynchronously
103        if let Some(parent) = path.parent() {
104            fs::create_dir_all(parent).await?;
105        }
106
107        let json_str = serde_json::to_string_pretty(&self.config)
108            .map_err(|e| std::io::Error::new(std::io::ErrorKind::Other, e))?;
109        fs::write(&path, json_str).await?;
110        Ok(())
111    }
