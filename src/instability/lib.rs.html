1//! This crate provides attribute macros for specifying API stability of public API items of a
2//! crate.
3//!
4//! The Rust standard library has a concept of [API stability] and custom attributes for managing
5//! that on a per-item basis, but most of these attributes are not available for normal crates to
6//! use, with the exception of the [`deprecated`] attribute. This crate seeks to provide similar
7//! attributes on stable Rust, though tuned more toward what the needs of normal crate authors.
8//!
9//! For complete examples of how to use this crate, check out the source code for the
10//! [`instability-example`] crate in the repository
11//!
12//! Currently, only the [`unstable`] attribute is available. Please see the documentation of that
13//! macro for an explanation on what it does and how to use it.
14//!
15//! [API stability]: https://rustc-dev-guide.rust-lang.org/stability.html
16//! [`deprecated`]:
17//!     https://doc.rust-lang.org/reference/attributes/diagnostics.html#the-deprecated-attribute
18//! [`instability-example`]: https://github.com/ratatui-org/instability/tree/master/example
19//! [`unstable`]: macro@unstable
20
21use proc_macro::TokenStream;
22use syn::{parse_macro_input, Item};

24mod unstable;

26/// Mark an API as unstable.
27///
28/// You can apply this attribute to an item in your public API that you would like to expose to
29/// users, but are not yet ready for general use. This is useful when you want to let users try out
30/// some new functionality for an API you haven't finished testing or designing, or for whatever
31/// reason do not want to commit any stability guarantees for.
32///
33/// This attribute does the following things to annotated items:
34///
35/// - Changes the visibility of the item from `pub` to `pub(crate)` unless a certain crate feature
36///   is enabled. This ensures that internal code within the crate can always use the item, but
37///   downstream consumers cannot access it unless they opt-in to the unstable API.
38/// - Changes the Visibility of certain child items of the annotated item (such as struct fields) to
39///   match the item's visibility. Children that are not public will not be affected.
40/// - Appends an "Availability" section to the item's documentation that notes that the item is
41///   unstable and indicates the name of the crate feature to enable it.
42///
43/// Child items of annotated modules are unchanged, as it might be desirable to be able to re-export
44/// them even if the module visibility is restricted. You should apply the attribute to each item
45/// within the module with the same feature name if you want to restrict the module's contents
46/// itself and not just the module namespace.
47///
48/// Note that unlike the [`unstable`][std-unstable] attribute used in the standard library, this attribute does
49/// not apply itself recursively to child items.
50///
51/// [std-unstable]: https://rustc-dev-guide.rust-lang.org/stability.html
52///
53/// Applying this attribute to non-`pub` items is pointless and does nothing.
54///
55/// # Arguments
56///
57/// The `unstable` attribute supports optional arguments that can be passed to control its behavior.
58///
59/// - `feature`: the name of the unstable feature that should control this item's availability. This
60///   will have the string `unstable-` prepended to it. If not specified, the item will instead be
61///   guarded by a catch-all `unstable` feature.
62/// - `issue`: a link or reference to a tracking issue for the unstable feature. This will be
63///   included in the item's documentation.
64///
65/// # Examples
66///
67/// We can apply the attribute to a public function like so:
68///
69/// ```
70/// /// This function does something really risky!
71/// ///
72/// /// Don't use it yet!
73/// #[instability::unstable(feature = "risky-function")]
74/// pub fn risky_function() {
75///     unimplemented!()
76/// }
77/// ```
78///
79/// This will essentially be expanded to the following:
80///
81/// ```
82/// /// This function does something really risky!
83/// ///
84/// /// Don't use it yet!
85/// ///
86/// /// # Availability
87/// ///
88/// /// **This API is marked as unstable** and is only available when the `unstable-risky-function`
89/// /// crate feature is enabled. This comes with no stability guarantees, and could be changed or
90/// /// removed at any time.
91/// #[cfg(feature = "unstable-risky-function")]
92/// pub fn risky_function() {
93///     unimplemented!()
94/// }
95///
96/// /// This function does something really risky!
97/// ///
98/// /// Don't use it yet!
99/// #[cfg(not(feature = "unstable-risky-function"))]
100/// pub(crate) fn risky_function() {
101///     unimplemented!()
102/// }
103/// ```
104#[proc_macro_attribute]
105pub fn unstable(args: TokenStream, input: TokenStream) -> TokenStream {
106    let mut attributes = unstable::UnstableAttribute::default();
107    let attributes_parser = syn::meta::parser(|meta| attributes.parse(meta));
108    parse_macro_input!(args with attributes_parser);
109
110    match parse_macro_input!(input as Item) {
111        Item::Type(item_type) => attributes.expand(item_type),
112        Item::Enum(item_enum) => attributes.expand(item_enum),
113        Item::Struct(item_struct) => attributes.expand(item_struct),
114        Item::Fn(item_fn) => attributes.expand(item_fn),
115        Item::Mod(item_mod) => attributes.expand(item_mod),
116        Item::Trait(item_trait) => attributes.expand(item_trait),
117        Item::Const(item_const) => attributes.expand(item_const),
118        Item::Static(item_static) => attributes.expand(item_static),
119        Item::Use(item_use) => attributes.expand(item_use),
120        _ => panic!("unsupported item type"),
121    }
122}
